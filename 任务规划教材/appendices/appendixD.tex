def nearest_neighbor(depot, customers, vehicle_capacity):
    """最近邻启发式"""
    routes = []
    unvisited = set(customers)
    while unvisited:
        route = [depot]
        load = 0
        while unvisited:
            nearest = find_nearest(route[-1], unvisited)
            if load + demand[nearest] <= vehicle_capacity:
                route.append(nearest)
                load += demand[nearest]
                unvisited.remove(nearest)
            else:
                break
        route.append(depot)
        routes.append(route)
    return routes
\end{lstlisting}
